【必备过滤】SQL 必须包含：
1. taxpayer_id = :taxpayer_id
2. 期间过滤（按用户问题选择）：
   - 年度：period_year = :year

【修订版本处理】
若需取最新修订版本，使用窗口函数：

年度申报：
WITH latest AS (
  SELECT taxpayer_id, taxpayer_name, period_year, revision_no,
         <需要查询的指标列>,
    ROW_NUMBER() OVER (
      PARTITION BY taxpayer_id, period_year
      ORDER BY revision_no DESC
    ) AS rn
  FROM <view>
  WHERE taxpayer_id = :taxpayer_id AND period_year = :year
)
SELECT <columns> FROM latest WHERE rn = 1

季度预缴：
WITH latest AS (
  SELECT taxpayer_id, taxpayer_name, period_year, period_quarter, revision_no,
         <需要查询的指标列>,
    ROW_NUMBER() OVER (
      PARTITION BY taxpayer_id, period_year, period_quarter
      ORDER BY revision_no DESC
    ) AS rn
  FROM <view>
  WHERE taxpayer_id = :taxpayer_id AND period_year = :year AND period_quarter = :quarter
)
SELECT <columns> FROM latest WHERE rn = 1

多季度查询（如"各季度"/"每个季度"）：
WITH latest AS (
  SELECT taxpayer_id, taxpayer_name, period_year, period_quarter, revision_no,
         <需要查询的指标列>,
    ROW_NUMBER() OVER (
      PARTITION BY taxpayer_id, period_year, period_quarter
      ORDER BY revision_no DESC
    ) AS rn
  FROM <view>
  WHERE taxpayer_id = :taxpayer_id AND period_year = :year
    AND period_quarter IN (1, 2, 3, 4)
)
SELECT <columns> FROM latest WHERE rn = 1
ORDER BY period_quarter
LIMIT {max_rows}
